\section{Quick Maths}

\subsection{Signal Processing}

\begin{definition}[Dirac Delta]
  \(\delta(x) = \begin{cases}
    0 & x \neq 0 \\
    \text{undefined} & x = 0
  \end{cases} \ \text{s.t.} \ \int_{\R} \delta(x) \,dx = 1\)
\end{definition}

\begin{theorem}
  \(\forall f: \int_{\R} f(x)\delta(x - \lambda) \,dx = f(\lambda)\)
\end{theorem}
